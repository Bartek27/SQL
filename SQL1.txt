SELECT ISSUESLIST.NAME, ISSUES.SUMMARY
FROM ISSUES
JOIN ISSUESLIST ON ISSUELIST.ID = ISSUE.ISSYESLIST_ID;


SELECT USERS.NAME, USERS.LASTNAME, ISSUES.NAME
FROM USERS
JOIN ISSUES ON USERS.ID = ISSUES.ISSUESLIST_ID;


SELECT USERS.NAME, USERS.LASTNAME, COUNT (*) AS ISSUES_NUMBER
FROM ISSUES
GROUP BY NAME;
